    def trigger_consolidation(self, k: int = None):
        """
        触发记忆巩固过程。
        :param k: 指定要从短期记忆中提取并巩固的条目数量。
                  如果不传，默认使用配置中的 memory_consolidation_threshold。
        """
        agent_name_log_prefix = f"[智能体 '{self.stes.agent_name}' 的记忆巩固引擎]"

        # 1. 确定要获取多少条记忆
        # 如果传入了 k (例如来自 Agent 的计数器)，就用 k；否则用默认阈值
        # 我们取 k 和 默认阈值 中的较大值，以防万一计数器出错导致 k 太小
        num_to_fetch = k if k is not None else memory_consolidation_threshold

        # 确保至少取 memory_consolidation_threshold 条，防止漏掉
        if num_to_fetch < memory_consolidation_threshold:
            num_to_fetch = memory_consolidation_threshold

        logger.info(f"\n{agent_name_log_prefix} - 步骤 1: 智能采样 (计划获取最近 {num_to_fetch} 条)")

        # 2. 使用动态的 k 去获取记忆
        most_recent_memories = self.stes.get_most_recent(k=num_to_fetch)

        # 检查是否有记忆
        if not most_recent_memories:
            logger.warning(f"{agent_name_log_prefix} 短期记忆为空，跳过巩固。")
            return


        theme_generation_prompt = f"""
        Based on the following recent events, generate a very short, concise core theme or question (preferably less than 50 words) to delve deeper into related memories.

        Recent events:
        {"\n".join(f"- {mem}" for mem in most_recent_memories)}

        Core topics/issues (very brief):
        """
        logger.info(f"{agent_name_log_prefix} 正在请求 LLM 生成反思主题...")
        try:
            theme_response = self.consolidation_llm.invoke(theme_generation_prompt)
            reflection_theme = theme_response.content.strip() if hasattr(theme_response, 'content') else str(
                theme_response).strip()
            logger.info(f"{agent_name_log_prefix} LLM 生成的反思主题 (原始): '{reflection_theme}'")

            # --- 【新增】检查并截断主题 ---
            if len(reflection_theme) > MAX_EMBEDDING_INPUT_CHARS:
                logger.warning(
                    f"{agent_name_log_prefix} 生成的主题过长 ({len(reflection_theme)} chars)，将截断至 {MAX_EMBEDDING_INPUT_CHARS} 字符。")
                reflection_theme = reflection_theme[:MAX_EMBEDDING_INPUT_CHARS]
                logger.info(f"{agent_name_log_prefix} 截断后的主题: '{reflection_theme}'")
            # --- 截断结束 ---

            logger.info(f"{agent_name_log_prefix} 正在根据主题 '{reflection_theme}' 搜索所有相关的历史记忆...")
            relevant_memories = self.retrieve_episodic_memories(reflection_theme, k=related_memories)  # 使用了已可能截断的主题

            combined_memories = most_recent_memories + relevant_memories #近期记忆里拿了最近阈值数量的记忆加上和这几个事件主题相关的记忆组合
            # 使用 set 去重，保持原始顺序（近似）
            unique_memories_set = set()
            unique_memories = []
            for mem in combined_memories:
                if mem not in unique_memories_set:
                    unique_memories.append(mem)
                    unique_memories_set.add(mem)

            if not unique_memories:
                logger.warning(f"{agent_name_log_prefix} 未找到足够的相关记忆进行反思，跳过巩固。")
                return

            logger.info(
                f"{agent_name_log_prefix} 从 STES 中采样了 {len(unique_memories)} 条独特的近期与相关记忆用于反思:")
            for mem in unique_memories:
                logger.debug(f"  - {mem}")  # 详细记忆内容改为 debug 级别

            # --- 【修改处】在反思前，获取相关的旧信念 ---
            related_beliefs_str = self._fetch_related_beliefs(reflection_theme)
            # --- 结束修改 ---


            logger.info(f"\n{agent_name_log_prefix} - 步骤 2: 反思")
            memories_str = "\n".join(f"- {mem}" for mem in unique_memories)

            logger.info(f"{agent_name_log_prefix} - 步骤 1.5: 硬事实预扫描...")
            fact_scan_prompt = f"""
            Identify all specific numbers, dates, and proper names from the following observations.
            Output them as a comma-separated list of "Entity: Value" pairs.
            If no specific facts exist, output "None".

            Observations:
            {memories_str}
            """
            try:
                fact_scan_response = self.consolidation_llm.invoke(fact_scan_prompt)
                hard_facts = fact_scan_response.content.strip()
                logger.info(f"{agent_name_log_prefix} 扫描到的硬事实: {hard_facts}")

                # 将扫描到的事实强制注入到接下来的反思提示词中
                memories_str = f"[[CRITICAL FACTS FOUND: {hard_facts}]]\n" + memories_str
            except Exception as e:
                logger.error(f"事实预扫描失败: {e}")
            # --- 【修改处】将旧信念和新证据都加入提示词 ---
#
            # 1. 强制温度为 0 的实现方式：
            # 在 trigger_consolidation 开始时，我们可以创建一个临时的低随机性实例
            # 如果你使用的是 LangChain 风格的 LLM 对象：
            consolidation_llm_low_temp = self.consolidation_llm.bind(temperature=0.0)

            # 2. 修改反思提示词
            reflection_prompt = f"""
            You are a logical deduction engine for a cognitive agent. Your task is to analyze recent evidence and past beliefs to generate **Logical Insights**, not abstract philosophy.

            **Step 1: HARD FACT EXTRACTION (Mandatory)**
            - Extract every specific number, date, duration, price, and proper name.
            - Example: "3.83 dollars", "14 days", "Pennsylvania".
            - DO NOT summarize or generalize these. Use exact words.

            **Step 2: LOGICAL DEDUCTION & INSIGHTS**
            Analyze the data for:
            1. **Social Relations**: (e.g., Based on A helping B, A likely trusts B).
            2. **Behavioral Patterns**: (e.g., Every time C mentions rice, they choose short-grain).
            3. **Causal Links**: (e.g., The milestone happened because the contract was signed).
            4. **Conflict Resolution**: If new evidence contradicts old beliefs, explain why the new data is more reliable (e.g., source is ground_truth).

            **Past Beliefs (from long-term memory):**
            {related_beliefs_str}

            **New Evidence (from short-term memory):**
            {memories_str}

            **Output Format:**
            [FACTS]: 
            - Fact 1 (e.g., Price is 3.83)
            - Fact 2 (e.g., Duration is 1 week)
            [INSIGHTS]:
            - Insight 1 (e.g., Tom is efficient because he completed the task in 1 week)
            - Insight 2 (e.g., The business is growing because a first client was signed)
            """

            # --- 结束修改 ---

            logger.info(f"{agent_name_log_prefix} 正在请求 LLM 生成反思...")
            reflection_response = self.consolidation_llm.invoke(reflection_prompt)
            reflection_text = reflection_response.content.strip() if hasattr(reflection_response, 'content') else str(
                reflection_response).strip()
            logger.info(f"{agent_name_log_prefix} LLM 生成的（更新后）反思:\n{reflection_text}")

            if not reflection_text:
                logger.warning(f"{agent_name_log_prefix} LLM 未能生成反思内容，跳过后续步骤。")
                return

            logger.info(f"\n{agent_name_log_prefix} - 步骤 3: 抽象与结构化")
            try:
                structured_llm = get_llm().with_structured_output(KnowledgeGraphExtraction, method="json_mode")
            except Exception as e:
                logger.error(f"{agent_name_log_prefix} 初始化结构化 LLM 时出错: {e}", exc_info=True)
                return  # 无法继续

            reflection_response = consolidation_llm_low_temp.invoke(reflection_prompt)
            reflection_text = reflection_response.content.strip()

            # 逻辑分层：确保提取器知道哪部分是死数据，哪部分是推理
            fact_part = ""
            insight_part = ""
            if "[FACTS]:" in reflection_text and "[INSIGHTS]:" in reflection_text:
                parts = reflection_text.split("[INSIGHTS]:")
                fact_part = parts[0].replace("[FACTS]:", "").strip()
                insight_part = parts[1].strip()

            # 1. 强制低随机性 LLM
            consolidation_llm_low_temp = self.consolidation_llm.bind(temperature=0.0)

            # 2. 优化后的提取提示词
            extraction_prompt = f"""
            IMPORTANT: You MUST output the result in valid JSON format in ENGLISH.
            You are a Knowledge Graph Specialist. Transform the "Updated Reflection" into a graph.

            **[SPECIAL NODE RULES]**:
            1. **Value/Date Nodes**: Mandatory for any number (3.83, 1 week).
            2. **Claim Nodes**: MANDATORY for statements made by entities. 
               - Example: If Tom says the password is '123', create a Claim node named "Password is 123".
            3. **Concept Nodes**: For high-level insights from the [INSIGHTS] section.

            **[SPECIAL RELATIONSHIP RULES]**:
            1. **Evidence Chain**: Connect `Concept` nodes to the `Value` or `Claim` nodes they are based on using `BASED_ON`.
            2. **Attribution**: (Person) -[PROPOSED]-> (Claim).
            3. **Claim Content**: (Claim) -[CONTENT]-> (Fact/Value/Entity).
            4. **Metadata**: Every relationship MUST have `source_of_belief` and `confidence`.

            **[LIE DETECTION]**:
            If the text reveals an entity said something they know is false, mark the relationship to the Claim as `{{"is_lie": true, "truth": "actual truth"}}`.

            Text to process:
            [[MANDATORY FACTS]]:
            {fact_part}
            [[LOGICAL DEDUCTIONS]]:
            {insight_part}
            """
            # --- 【核心修改结束】 ---

            # =========================================================================
            # 【深度优化版】手动解析、清洗并修补知识图谱数据
            # 【工业级优化版】手动解析、清洗并修补知识图谱数据
            # =========================================================================
            logger.info(f"{agent_name_log_prefix} 正在请求 LLM 将反思结构化为知识图谱...")

            try:
                response = consolidation_llm_low_temp.invoke(extraction_prompt)
                llm_output = response.content.strip()
            except Exception as e:
                logger.error(f"{agent_name_log_prefix} LLM 调用失败: {e}")
                return

            # --- 1. 强化版 JSON 提取与预清洗 ---
            data_dict = {"nodes": [], "relationships": []}
            json_str = ""
            try:
                # 策略 A: 尝试代码块提取
                match = re.search(r'```json\s*(\{.*?\})\s*```', llm_output, re.DOTALL)
                if match:
                    json_str = match.group(1)
                else:
                    # 策略 B: 贪婪匹配最外层花括号 (应对截断情况)
                    start = llm_output.find("{")
                    end = llm_output.rfind("}")
                    if start != -1 and end != -1:
                        json_str = llm_output[start:end + 1]

                if not json_str:
                    raise ValueError("未能识别到 JSON 结构")

                # 深度清洗：处理常见的格式瑕疵
                json_str = json_str.replace('\n', ' ').replace('\r', ' ')  # 消除非法换行
                json_str = re.sub(r'//.*?$', '', json_str, flags=re.MULTILINE)  # 移除可能存在的注释
                json_str = re.sub(r',\s*\}', '}', json_str)  # 修复末尾多余逗号
                json_str = re.sub(r',\s*\]', ']', json_str)  # 修复数组末尾多余逗号

                data_dict = json.loads(json_str)
            except Exception as e:
                logger.error(f"{agent_name_log_prefix} JSON 最终解析失败: {e} | 片段: {llm_output[:100]}...")
                return

            # --- 2. 核心修复逻辑：标签归一化与查找表 ---
            nodes_list = data_dict.get("nodes", [])
            rels_list = data_dict.get("relationships", [])
            name_to_label = {}
            unique_nodes = {}

            for n in nodes_list:
                if not isinstance(n, dict) or "name" not in n: continue

                name_val = n["name"].lower()
                # 自动修正逻辑：如果名字里包含时间单位或数字，强制设为 Value/Duration
                time_units = ["hours", "days", "mins", "weeks", "months", "years"]
                # 检查是否包含数字，或者包含特定的时间单位单词
                has_digit = any(char.isdigit() for char in name_val)
                has_time_unit = any(re.search(rf"\b{unit}\b", name_val) for unit in time_units)

                if has_digit or has_time_unit:
                    if n.get("label") == "Concept":
                        n["label"] = "Value"

                # 【优化点：标签归一化】强制首字母大写，防止 Neo4j 匹配失效
                raw_label = n.get("label", "Concept")
                normalized_label = raw_label.strip().capitalize()  # person -> Person
                n["label"] = normalized_label

                name_key = n["name"].strip()
                name_to_label[name_key] = normalized_label
                unique_nodes[name_key] = n

            # --- 3. 关系修补与“孤儿节点”自动召回 ---
            valid_rels = []
            seen_rel_keys = set()

            for rel in rels_list:
                if not isinstance(rel, dict): continue

                s_name = rel.get("source_node_name", "").strip()
                t_name = rel.get("target_node_name", "").strip()
                r_type = rel.get("type", "RELATED_TO").upper()  # 关系名强制大写

                if not s_name or not t_name: continue

                # 【优化点：自动补全缺失节点逻辑】
                for role in ["source", "target"]:
                    name_val = s_name if role == "source" else t_name
                    label_key = f"{role}_node_label"

                    # 如果关系里没写 label，或者节点表里根本没这个点
                    if not rel.get(label_key) or name_val not in name_to_label:
                        # 1. 优先从已有节点表里查
                        if name_val in name_to_label:
                            rel[label_key] = name_to_label[name_val]
                        else:
                            # 2. 查不到则根据名字猜测，或默认设为 Concept
                            # 比如：如果是数字，设为 Value；否则 Concept
                            guessed_label = "Value" if any(char.isdigit() for char in name_val) else "Concept"
                            rel[label_key] = guessed_label
                            # 3. 重要：自动在节点表里补上这个点，否则 Neo4j 写入会失败
                            if name_val not in unique_nodes:
                                unique_nodes[name_val] = {
                                    "name": name_val,
                                    "label": guessed_label,
                                    "properties": {"source_of_belief": "auto_recovered"}
                                }
                                name_to_label[name_val] = guessed_label

                # 关系去重
                rel_fingerprint = f"{s_name}-{r_type}-{t_name}"
                if rel_fingerprint not in seen_rel_keys:
                    valid_rels.append(rel)
                    seen_rel_keys.add(rel_fingerprint)

            # --- 4. 最终数据验证与构造 ---
            repaired_data = {"nodes": list(unique_nodes.values()), "relationships": valid_rels}
            try:
                structured_response = KnowledgeGraphExtraction.model_validate(repaired_data)
            except Exception as e:
                logger.warning(f"{agent_name_log_prefix} 结构校验失败，强制构造: {e}")
                structured_response = KnowledgeGraphExtraction.model_construct(**repaired_data)

            # =============================================================

            logger.info(f"{agent_name_log_prefix} LLM 已返回结构化数据:")
            try:
                # 使用 .model_dump_json() 获取 JSON 字符串，美化打印
                #TODO 这里的debug改成info就能查看json日志
                logger.debug(structured_response.model_dump_json(indent=2, ensure_ascii=False))
                # logger.info(structured_response.model_dump_json(indent=2, ensure_ascii=False))

            except Exception:  # 回退到打印对象本身
                logger.info(structured_response)

            logger.info(f"\n{agent_name_log_prefix} - 步骤 4: 整合到共享 LTSS")
            if self.ltss and self.ltss.driver:

                # --- 【论文创新点：增强写入的数据】 ---
                # 【修改】使用 timezone-aware 对象
                current_timestamp = datetime.datetime.now(datetime.UTC).isoformat()

                # --- 【优化 1：创建 TextUnit 证据锚点】 ---
                # 这借鉴了 GraphRAG，让所有新知识都有据可查
                text_unit_id = f"unit_{datetime.datetime.now().strftime('%Y%m%d%H%M%S')}"
                self.ltss.update_graph(
                    "MERGE (u:TextUnit {name: $name}) SET u.content = $content, u.timestamp = $ts",
                    {"name": text_unit_id, "content": memories_str, "ts": current_timestamp}
                )

                if structured_response.relationships:
                    rels_data = []
                    for rel in structured_response.relationships:
                        props = rel.properties if isinstance(rel.properties, dict) else {}

                        # --- 【创新点：细化元数据分类】 ---
                        # 区分：原生观察事实 vs 逻辑推理产物
                        if rel.type == "BASED_ON" or "Concept" in [rel.source_node_label, rel.target_node_label]:
                            props["knowledge_type"] = "logical_inference"
                        else:
                            props["knowledge_type"] = "observed_fact"

                        # 注入元数据
                        props["consolidated_at"] = current_timestamp
                        props["evidence_source_unit"] = text_unit_id  # 挂载证据链 ID

                        if "source_of_belief" not in props:
                            props["source_of_belief"] = "reflection"
                        if "confidence" not in props:
                            props["confidence"] = 0.9

                        rels_data.append({
                            "source_name": rel.source_node_name,
                            "source_label": rel.source_node_label,
                            "target_name": rel.target_node_name,
                            "target_label": rel.target_node_label,
                            "type": rel.type,
                            "props": props
                        })

                        # --- 【优化 2：建立知识到证据的物理连接】 ---
                        # 让这个关系中的起始节点指向 TextUnit 节点
                        # (这在图谱可视化时会非常漂亮，呈现放射状指向原始文本)
                        self.ltss.update_graph(
                            f"MATCH (n:`{rel.source_node_label}` {{name: $nname}}), (u:TextUnit {{name: $uname}}) "
                            "MERGE (n)-[:FROM_SOURCE]->(u)",
                            {"nname": rel.source_node_name, "uname": text_unit_id}
                        )

                # 【修改】优化：批量写入节点 (Batch Node Insertion)
                # =========================================================================
                if structured_response.nodes:
                    logger.info(f"{agent_name_log_prefix} 正在批量处理 {len(structured_response.nodes)} 个新节点...")

                    nodes_to_embed = []
                    node_data_list = []

                    for node in structured_response.nodes:
                        if not isinstance(node.properties, dict):
                            node.properties = {}

                        # 默认权限
                        if "accessible_by" not in node.properties:
                            node.properties["accessible_by"] = [self.agent_name]

                        # 准备 Embedding 文本
                        desc = node.properties.get('description', '')
                        text = f"{node.label}: {node.name}. {desc}"
                        nodes_to_embed.append(text)

                        # 暂存数据
                        node_data_list.append({
                            "name": node.name,
                            "label": node.label,
                            "props": node.properties
                        })

                    # --- 1. 批量生成向量 (1次 API 请求) ---
                    try:
                        if nodes_to_embed:
                            vectors = self.embedding_model.embed_documents(nodes_to_embed)
                            for i, vector in enumerate(vectors):
                                node_data_list[i]["props"]["embedding"] = vector
                    except Exception as e:
                        logger.error(f"批量生成向量失败: {e}")

                    # --- 2. 批量写入 Neo4j (按 Label 分组写入) ---
                    nodes_by_label = {}
                    for data in node_data_list:
                        label = data["label"]
                        if label not in nodes_by_label:
                            nodes_by_label[label] = []
                        nodes_by_label[label].append(data)

                    for label, batch_data in nodes_by_label.items():
                        cypher_query = f"""
                                        UNWIND $batch AS row
                                        MERGE (n:`{label}` {{name: row.name}})
                                        SET n += row.props
                                        """
                        try:
                            self.ltss.update_graph(cypher_query, parameters={"batch": batch_data})
                            logger.info(f"  > 批量写入 {len(batch_data)} 个 '{label}' 节点成功。")
                        except Exception as e:
                            logger.error(f"批量写入 '{label}' 节点失败: {e}")
                # =======================================

                # 【优化版】原项目逻辑：批量准备数据 + 快速写入关系
                # =========================================================================
                if structured_response.relationships:
                    logger.info(
                        f"{agent_name_log_prefix} 正在批量处理 {len(structured_response.relationships)} 条关系 (保留信念元数据)...")

                    rels_data = []
                    for rel in structured_response.relationships:
                        props = rel.properties if isinstance(rel.properties, dict) else {}

                        # 创新点：区分“原生事实”与“推论事实”
                        # 如果关系是从 [INSIGHTS] 提取的，标记为 inference
                        if rel.type == "BASED_ON" or "Insight" in rel.source_node_label:
                            props["knowledge_type"] = "logical_inference"
                        else:
                            props["knowledge_type"] = "observed_fact"

                        # 记录观测到的最早/最晚时间戳，而不仅仅是巩固时间
                        props["observed_time_range"] = [current_timestamp]  # 后续可以合并

                        # --- 【原项目逻辑：保留完整的元数据计算】 ---

                        # 1. 知识形成时间
                        props["consolidated_at"] = current_timestamp

                        # 2. 知识来源 (如果未提取，默认为反思)
                        if "source_of_belief" not in props:
                            props["source_of_belief"] = "reflection"

                        # 3. 置信度处理 (如果未提取，默认为 0.9)
                        if "confidence" not in props:
                            props["confidence"] = 0.9
                        else:
                            try:
                                props["confidence"] = float(props["confidence"])
                            except (ValueError, TypeError):
                                props["confidence"] = 0.9

                        # ----------------------------------------

                        rels_data.append({
                            "source_name": rel.source_node_name,
                            "source_label": rel.source_node_label,
                            "target_name": rel.target_node_name,
                            "target_label": rel.target_node_label,
                            "type": rel.type,
                            "props": props
                        })

                    # --- 快速写入阶段 ---
                    # 将 IO 操作与逻辑计算分离，循环速度会更快
                    success_count = 0
                    for rel_data in rels_data:
                        cypher_query = f"""
                            MERGE (source:`{rel_data['source_label']}` {{name: $source_name}})
                            MERGE (target:`{rel_data['target_label']}` {{name: $target_name}})
                            MERGE (source)-[r:`{rel_data['type']}`]->(target)
                            SET r = $props
                            """
                        try:
                            self.ltss.update_graph(cypher_query, parameters=rel_data)
                            success_count += 1
                        except Exception as e:
                            logger.error(f"关系写入失败: {e}")

                    logger.info(
                        f"{agent_name_log_prefix} 成功写入 {success_count}/{len(rels_data)} 条关系。")
                # ==================================================
            else:
                logger.warning(f"{agent_name_log_prefix} LTSS 不可用，跳过写入数据库。")

            logger.info(f"{agent_name_log_prefix} 记忆巩固流程完成。")

        except Exception as e:
            # 捕获整个巩固过程中的其他意外错误
            logger.error(f"{agent_name_log_prefix} 记忆巩固过程中发生意外错误: {e}", exc_info=True)
